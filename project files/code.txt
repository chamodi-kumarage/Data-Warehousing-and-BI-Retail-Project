CREATE TABLE dbo.FactSales
(
    SalesFactKey            BIGINT          IDENTITY(1,1) NOT NULL,
    transaction_id          NVARCHAR(36)    NOT NULL,
    TransactionDateKey      INT             NOT NULL,
    CustomerKey             INT             NOT NULL,
    ProductKey              INT             NOT NULL,
    StoreKey                INT             NOT NULL,
    PaymentMethodKey        INT             NOT NULL,
    quantity                INT             NULL,
    price                   DECIMAL(18,2)   NULL,
    discount_applied        DECIMAL(5,2)    NULL,
    accm_txn_create_time    DATETIME2(0)    NOT NULL,
    accm_txn_complete_time  DATETIME2(0)    NULL,
    txn_process_time_hours  DECIMAL(18,2)   NULL,
    LoadDTS                 DATETIME2(0)    NOT NULL DEFAULT SYSDATETIME(),
    CONSTRAINT PK_FactSales PRIMARY KEY (SalesFactKey),
    CONSTRAINT UQ_FactSales UNIQUE (transaction_id),
    CONSTRAINT FK_FactSales_Date FOREIGN KEY (TransactionDateKey) REFERENCES dbo.DimDate(DateKey),
    CONSTRAINT FK_FactSales_Customer FOREIGN KEY (CustomerKey) REFERENCES dbo.DimCustomer(CustomerKey),
    CONSTRAINT FK_FactSales_Product FOREIGN KEY (ProductKey) REFERENCES dbo.DimProduct(ProductKey),
    CONSTRAINT FK_FactSales_Store FOREIGN KEY (StoreKey) REFERENCES dbo.DimStore(StoreKey),
    CONSTRAINT FK_FactSales_Payment FOREIGN KEY (PaymentMethodKey) REFERENCES dbo.DimPaymentMethod(PaymentMethodKey)
);
GO
